37bool GEU::ThermalAugment::ThermalImageParams::registerAlignedImage(const ThermalAugment& augmenter)
38{
39 // static constexpr unsigned NUM_ECC_ITERATIONS = 60;
40 // static constexpr float ECC_PRECISION = 1e-1f;
41 #define NUM_ECC_ITERATIONS 60
42 #define ECC_PRECISION 1e-1
43 #define RECTANGLE_THRESHOLD vec2(20.0f, 20.0f)
44 cv::utils::logging::setLogLevel(cv::utils::logging::LogLevel::LOG_LEVEL_INFO);
45
46 bool success = false;
47
48 const vec2 shrinkK = augmenter._globalCamParamsRGB._imageCropScale;
49 const vec2 offset = augmenter._globalCamParamsRGB._imageCropOffset;
50 vec2 rgbSize = _size, thermalSize = _alignedImage->_size;
51 vec2 center = rgbSize / 2.0f + offset;
52
53 vec2 newSize = vec2(rgbSize.x * shrinkK.x, rgbSize.y * shrinkK.y);
54 vec2 newSize_2 = newSize / 2.0f;
55 vec2 shrinkSize = thermalSize;
56 vec2 raspect = newSize / shrinkSize;
57 cv::Rect cropRectangle = cv::Rect(center.x - newSize_2.x, center.y - newSize_2.y, newSize.x, newSize.y);
58 cv::Mat warpMatrix;
59
60 OpenCVImage* grayscaleRGBImage = _rawImage->convertToGrayscale();
61 OpenCVImage* grayscaleThermalImage = _alignedImage->_rawImage->convertToGrayscale();
62
63 OpenCVImage* croppedImage = grayscaleRGBImage->crop(cropRectangle);
64 OpenCVImage* shrinkedImage_rgb = croppedImage->resize(shrinkSize);
65 OpenCVImage* shrinkedImage_thermal = grayscaleThermalImage->resize(shrinkSize);
66
67 bool alignedThermal = shrinkedImage_rgb->enhancedCorrelationCoefficient(shrinkedImage_thermal, cv::MOTION_HOMOGRAPHY, NUM_ECC_ITERATIONS, ECC_PRECISION, warpMatrix);
68
69 // Compose rectangle and apply transformations if images were aligned
70 if (alignedThermal) {//shrinkedImage_rgb->enhancedCorrelationCoefficient(shrinkedImage_thermal, cv::MOTION_HOMOGRAPHY, NUM_ECC_ITERATIONS, ECC_PRECISION, warpMatrix)) {
71 auto area = new Rectangle2D(vec2(.0f), vec2(shrinkSize.x, .0f), vec2(.0f, shrinkSize.y), shrinkSize); // Initial area
72 auto rotatedArea = OpenCVImage::composeRectangle(area, warpMatrix);
73 rotatedArea->translateCorners(-shrinkSize / 2.0f);
74 rotatedArea->scaleCorners(raspect);
75 rotatedArea->translateCorners(center); // From (0, 0) to center
76
77 if (!rotatedArea->exceedsSlope(RECTANGLE_THRESHOLD.x, RECTANGLE_THRESHOLD.y)) {
78
79 // Build matrix to get the inverse process
80 {
81 cv::Mat invMatrix = warpMatrix.inv();
82 auto glmInvMatrix = mat3(1.0f);
83
84 // Fill warp matrix of imageType with the result: glm has columnwise access, while OpenCV access first to the rows
85 for (int row = 0; row < 3; ++row) {
86 for (int col = 0; col < 3; ++col) {
87 glmInvMatrix[row][col] = invMatrix.at<float>(cv::Point(row, col));
88 }
89 }
90
91 _warpMatrix = glmInvMatrix * glm::translate(mat3(1.0f), shrinkSize / 2.0f) * glm::scale(mat3(1.0f), 1.0f / raspect) * glm::translate(mat3(1.0f), -center);
92 }
93
94 success = true;
95 }
96
97 delete area;
98 delete rotatedArea;
99 }
100
101 // Delete pointers to images
102 delete grayscaleRGBImage;
103 delete grayscaleThermalImage;
104 delete croppedImage;
105 delete shrinkedImage_rgb;
106 delete shrinkedImage_thermal;
107
108 _registered = success;
109 return success;
110}

147void GEU::ThermalAugment::readCameraCalibratedParameters(const std::string& filename, std::map<std::string, ThermalCameraParams>& targetTypeMap) {
148 std::string line;
149 std::ifstream in;
150 std::vector<std::string> stringTokens;
151 std::vector<float> floatTokens;
152 unsigned index = -1, translationOffset = 0, numberOfLines;
153 bool foundImageHeader = false, foundTranslationHeader = false;
154 std::vector<std::string> lines;
155
156
157 in.open(filename);
158 if (!in) {
159 throw std::exception();
160 }
161
162 while (!(in >> std::ws).eof() && !foundImageHeader) {
163 std::getline(in, line);
164 lines.push_back(line);
165 }
166
167 while (++index < lines.size() && !foundImageHeader) {
168 const size_t xIndex = lines[index].find_last_of("x");
169
170 try {
171 numberOfLines = (xIndex != std::string::npos) ? std::stoi(std::string(1, lines[index][xIndex + 1])) : 1;
172 }
173 catch (std::exception&) {
174 numberOfLines = 1;
175 }
176
177 if (!foundTranslationHeader) {
178 if (lines[index].find(CAMERA_TRANSLATION_HEADER) != std::string::npos) {
179 foundTranslationHeader = true;
180 }
181 else {
182 translationOffset += numberOfLines;
183 }
184 }
185
186 foundImageHeader = lines[index].find("DJI") != std::string::npos || lines[index].find("dji") != std::string::npos;
187 }
188
189 --index; // Go back to DJI line
190
191 while (index < lines.size()) {
192 FileManagement::clearTokens(stringTokens, floatTokens);
193 FileManagement::readTokens(lines[index++], DELIMITER, stringTokens, floatTokens);
194
195 auto cameraName = stringTokens[0];
196 cameraName = cameraName.substr(0, cameraName.find_last_of('.'));
197
198 auto imageSize = ivec2(floatTokens[0], floatTokens[1]);
199
200 // Camera matrix K
201 FileManagement::clearTokens(stringTokens, floatTokens);
202 FileManagement::readTokens(lines[index++], DELIMITER, stringTokens, floatTokens);
203 FileManagement::readTokens(lines[index++], DELIMITER, stringTokens, floatTokens);
204 FileManagement::readTokens(lines[index++], DELIMITER, stringTokens, floatTokens);
205
206 auto matrixK = floatTokens;
207
208 // Radial distortion
209 FileManagement::clearTokens(stringTokens, floatTokens);
210 FileManagement::readTokens(lines[index++], DELIMITER, stringTokens, floatTokens);
211
212 auto radialDist = vec3(floatTokens[0], floatTokens[1], floatTokens[2]);
213
214 // Tangential distortion
215 FileManagement::clearTokens(stringTokens, floatTokens);
216 FileManagement::readTokens(lines[index++], DELIMITER, stringTokens, floatTokens);
217
218 auto tangDist = vec2(floatTokens[0], floatTokens[1]);
219
220 // Translation
221 FileManagement::clearTokens(stringTokens, floatTokens);
222 FileManagement::readTokens(lines[index++], DELIMITER, stringTokens, floatTokens);
223
224 auto localPos = vec3(floatTokens[0], floatTokens[1], floatTokens[2]);
225
226 // Matrix
227 FileManagement::clearTokens(stringTokens, floatTokens);
228 FileManagement::readTokens(lines[index++], DELIMITER, stringTokens, floatTokens);
229 FileManagement::readTokens(lines[index++], DELIMITER, stringTokens, floatTokens);
230 FileManagement::readTokens(lines[index++], DELIMITER, stringTokens, floatTokens);
231
232 auto rotMatrix = floatTokens;
233
234 targetTypeMap[cameraName] = ThermalCameraParams{
235 ._localPosition = localPos,
236 ._omegaPhiKappa = vec3(0.0f),
237 ._pMatrix = mat4(1.0f),
238 ._rotationMatrix = glm::transpose(glm::make_mat3(rotMatrix.data())),
239 ._utmPosition = vec3(0.0f),
240 ._imageParams = ThermalImageParams{
241 ._imageName = cameraName,
242 ._size = imageSize,
243 ._matrixK = glm::transpose(glm::make_mat3(matrixK.data())),
244 ._radialDistortion = radialDist,
245 ._tangentialDistortion = tangDist,
246 },
247
248 };
249 }

569void GEU::ThermalAugment::readTemperatureFLIR(const std::filesystem::path& thermalCSVpath, float& progress)
570{
571 static constexpr float PROGRESS_INC = (0.60f - 0.40f);
572
573 std::vector<std::filesystem::directory_entry> files;
574 for (const auto& csvFile : std::filesystem::directory_iterator{ thermalCSVpath })
575 {
576 if (!csvFile.is_regular_file() or !_camerasThermal.contains(csvFile.path().stem().string()))
577 {
578 continue;
579 }
580 files.push_back(csvFile);
581 }
582
583 const float adjusted_prog_incr = PROGRESS_INC / files.size();
584
585 for(int i = 0; i < files.size(); ++i){
586 auto it = files.begin();
587 std::advance(it, i);
588 auto csvFile = *it;
589
590 auto& cameraParams = _camerasThermal[csvFile.path().stem().string()];
591 std::string line;
592 std::ifstream in;
593 uvec2 imageSize = _globalCamParamsThermal._imageOriginalSize;
594 std::vector<std::string> stringTokens;
595 std::vector<float> floatTokens;

598 if (!in)

600 progress += adjusted_prog_incr;

605
606 while (!(in >> std::ws).eof()) {
607 std::getline(in, line);
608
609 FileManagement::clearTokens(stringTokens, floatTokens);
610 FileManagement::readTokens(line, ',', stringTokens, floatTokens);
611

618
619 {

621 auto image = new OpenCVImage(temperatureMatrix);

623 imageSize = undistortedImage->getSize();
624
625 cameraParams._imageParams._temperature.resize(imageSize.x * imageSize.y);
626 #ifdef _OPENMP
627 #pragma omp parallel for collapse(1)
628 #endif
629 for (unsigned row = 0; row < imageSize.y; ++row) {
630 for (unsigned col = 0; col < imageSize.x; ++col) {
631 cameraParams._imageParams._temperature[row * imageSize.x + col] = undistortedImage->getColor(row, col);
632 }

637 }

639 progress += adjusted_prog_incr;

756void GEU::ThermalAugment::augmentCloud(float& progress)
757{
758 static constexpr float PROGRESS_INC = (0.90f - 0.60f);
759
760 auto pointCloud = _pointCloud->getCloud();
761 AABB aabb = pointCloud->getAABB();
762 auto pointData = pointCloud->getPointsPosition();
763 float height = pointCloud->getAABB().center().z;
764 std::vector temperatureColor(pointCloud->getNumberOfPoints(), 0.0f);
765
766 const float adjusted_prog_incr = PROGRESS_INC / _camerasRGB.size();
767
768 #ifdef _OPENMP
769 #pragma omp parallel for
770 #endif
771 for (int cameraIdx = 0; cameraIdx < _camerasRGB.size(); ++cameraIdx)
772 {
773 auto cameraIt = _camerasRGB.begin();
774 std::advance(cameraIt, cameraIdx);
775 auto cameraRGB = cameraIt->second;
776 auto imageRGB = cameraRGB._imageParams;
777
778 if (cameraRGB._imageParams._alignedImage)
779 {
780 std::vector<uint32_t> indices;
781 _octree->radiusNeighbors<unibn::L2Distance<vec3>>(vec3(cameraRGB._localPosition.x, cameraRGB._localPosition.y, height), 20.0f, indices);
782 for (const auto& index : indices)
783 {
784 auto imgPoint = cameraRGB.transformTo2D(vec4(pointData.at(index), 1));
785 float pointTemperature;
786 if (imageRGB.retrieveTemperature(imgPoint, _rectangleThermal, pointTemperature))
787 {
788 temperatureColor[index] = pointTemperature;
789 }
790 }
791 }
792 progress += adjusted_prog_incr;
793 }
794
795
796 pointCloud->setFusionData(PointCloud::dataSource::THERMAL, 0, temperatureColor);
797}
