149 distortionCoefficients.at<float>(cv::Point2i(4, 0)) = radialDistortion[2];

151 cv::undistort(_image, result, K, distortionCoefficients);
152 cv::undistortPoints(pointCorners, undistortedPoints, K, distortionCoefficients, cv::noArray(), K);

154 OpenCVImage* undistortedImage = outcome = new OpenCVImage(result);
155 vec2 minPoint = vec2(std::max(undistortedPoints[0].x, undistortedPoints[2].x), std::max(undistortedPoints[0].y, undistortedPoints[1].y)),
156 maxPoint = vec2(std::min(undistortedPoints[1].x, undistortedPoints[3].x), std::min(undistortedPoints[2].y, undistortedPoints[3].y));
157 bool needCrop = !(minPoint.x <= .0f && minPoint.y <= .0f && maxPoint.x >= imageSize.x && maxPoint.y >= imageSize.y);
158
159 if (needCrop) {

161 cv::Rect cvCropRectangle = cv::Rect(minPoint.x, minPoint.y, rectangleSize.x, rectangleSize.y);

184vec3 OpenCVImage::getColor(const vec2& pixel) {
185 const vec2 imageSize(_image.cols, _image.rows);
186 const vec2 minusPixel = glm::clamp(pixel - vec2(0.5f - glm::epsilon<float>()), vec2(.0f), imageSize);
187 const vec2 plusPixel = glm::clamp(pixel + vec2(0.5f - glm::epsilon<float>()), vec2(.0f), imageSize);
188 const float u = fmod(pixel.x, (int)pixel.x), v = fmod(pixel.y, (int)pixel.y);
189
190 // Access pixels
191 const vec3 lowerLeft = this->getColor(ivec2(minusPixel.x, minusPixel.y));
192 const vec3 lowerRight = this->getColor(ivec2(plusPixel.x, minusPixel.y));
193 const vec3 topLeft = this->getColor(ivec2(minusPixel.x, plusPixel.y));
194 const vec3 topRight = this->getColor(ivec2(plusPixel.x, plusPixel.y));
195
196 // Bilinear interpolation
197 return (1.0f - u) * (1.0f - v) * lowerLeft + u * (1.0f - v) * lowerRight + (1.0f - u) * v * topLeft + u * v * topRight;
198}
