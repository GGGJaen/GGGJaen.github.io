82bool GEU::VAO::support(const ShaderEnum::ShaderTypes& shaderType, const uint8_t& subroutine) const {
83 switch (shaderType) {
84 case ShaderEnum::VERTEX_SHADER:
85 switch (subroutine) {
86 case ShaderEnum::VERTEX_COLOR_SOURCE:
87 return _defineVBO[VBO_COLOR_RGB] && (_defineVBO[VBO_THERMAL] || _defineVBO[VBO_MULTISPECTRAL] || _defineVBO[VBO_HYPERSPECTRAL]);
88 default:
89 return false;
90 }
91 case ShaderEnum::GEOMETRY_SHADER:
92 switch (subroutine) {
93 case ShaderEnum::DRAW_NORMALS:
94 return _defineVBO[VBO_NORMAL];
95 default:
96 return false;
97 }
98 case ShaderEnum::FRAGMENT_SHADER:
99 switch (subroutine) {
100 case ShaderEnum::LIGHTING:
101 return _defineVBO[VBO_NORMAL];
102 case ShaderEnum::LIGHT_ATTENUATION:
103 return _defineVBO[VBO_NORMAL];
104 case ShaderEnum::FRAGMENT_COLOR_SOURCE:
105 return _defineVBO[VBO_COLOR_RGB];
106 default:
107 return false;
108 }
109 default:
110 return false;
111 }
112}
