72inline vec3 RandomUtilities::getRandomToSphere(const float radius, const float distanceSquared)
73{
74 const float r1 = getUniformRandom();
75 const float r2 = getUniformRandom();
76 const float z = 1 + r2 * (sqrt(1.0f - radius * radius / distanceSquared) - 1.0f);
77 const float phi = 2.0f * glm::pi<float>() * r1;
78 const float x = std::cos(phi) * sqrt(1 - z * z);
79 const float y = std::sin(phi) * sqrt(1 - z * z);
80
81 return vec3(x, y, z);
82}
