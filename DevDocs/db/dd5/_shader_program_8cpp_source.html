113GLuint GEU::ShaderProgram::compileShader(const char* filename, const GLenum shaderType) {
114 if (!fileExists(filename)) {
115 if (shaderType != GL_GEOMETRY_SHADER)
116 spdlog::error("Shader source file {} not found", filename);
117
118 return 0;
119 }
120
121 std::string shaderSourceString;
122 if (!loadFileContent(std::string(filename), shaderSourceString)) { // Read shader code
123 spdlog::error("Could not open shader {} source file", filename);
124 return 0;
125 }
126
127 if (!includeLibraries(shaderSourceString)) { // Libraries code not found
128 spdlog::error("Could not include the specified modules");
129 return 0;
130 }
131
132 const GLuint shaderHandler = glCreateShader(shaderType);
133 if (shaderHandler == 0) {
134 spdlog::error("Could not create shader objects");
135 return 0;
136 }
137
138 const char* shaderSourceCString = shaderSourceString.c_str(); // Compile shader code
139 glShaderSource(shaderHandler, 1, &shaderSourceCString, nullptr);
140 glCompileShader(shaderHandler);
141
142 GLint compileResult = 0;
143 GLint logLen = 0;
144 _logString = "";
145 glGetShaderiv(shaderHandler, GL_COMPILE_STATUS, &compileResult); // Result
146 glGetShaderiv(shaderHandler, GL_INFO_LOG_LENGTH, &logLen);
147 if (logLen > 0) {
148 const auto cLogString = new char[logLen];
149 GLint written = 0;
150
151 glGetShaderInfoLog(shaderHandler, logLen, &written, cLogString);
152 _logString.assign(cLogString);
153
154 delete[] cLogString;
155 spdlog::warn("Shader log: {}", _logString);
156 }
157 spdlog::info("{} {}", shaderType, (compileResult == GL_FALSE ? " compiled failed" : " compiled sucessfully"));
158
159 return shaderHandler;
160}

170bool GEU::ShaderProgram::includeLibraries(std::string& shaderContent) {
171 size_t pos = shaderContent.find(MODULE_HEADER);
172
173 while (pos != std::string::npos) {
174 const std::size_t char_1 = shaderContent.find(MODULE_FILE_CHAR_1, pos);
175 const std::size_t char_2 = shaderContent.find(MODULE_FILE_CHAR_2, pos);
176
177 if ((char_1 == std::string::npos) || (char_2 == std::string::npos)) { // Incorrect syntax
178 return false;
179 }
180
181 const std::string module = shaderContent.substr(char_1 + 1, char_2 - char_1 - 1);
182 if (!fileExists(module)) { // Library refers to a new file, does it exist?
183 return false;
184 }
185
186 const auto moduleCode = _moduleCode.find(module); // If file is already read we can just retrieve the string
187 std::string moduleCodeStr;
188 if (moduleCode == _moduleCode.end()) {
189 if (!loadFileContent(module, moduleCodeStr)) {
190 return false;
191 }
192
193 _moduleCode[module] = moduleCodeStr;
194 } else {
195 moduleCodeStr = moduleCode->second;
196 }
197
198 shaderContent.erase(shaderContent.begin() + pos, shaderContent.begin() + char_2 + 1); // Replace string in shader code
199 shaderContent.insert(pos, moduleCodeStr);
200 pos = shaderContent.find(MODULE_HEADER);
201 }
202
203 return true;
204}
