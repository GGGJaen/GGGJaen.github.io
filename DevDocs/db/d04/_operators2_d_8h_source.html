89 const vec2 a3 = { rectACenter.x + rectAExtent.x, rectACenter.y - rectAExtent.y };

91 const vec2 b1 = { rectBCenter.x - rectBExtent.x, rectBCenter.y + rectBExtent.y };
92 const vec2 b2 = { rectBCenter.x + rectBExtent.x, rectBCenter.y + rectBExtent.y };

103 b1 - b0,
104 b2 - b1,
105 b3 - b2,
106 b0 - b3
107 };
108 std::vector<vec2> testAxes = {
109 {-rectsPointsA[0].y, rectsPointsA[0].x},
110 {-rectsPointsA[1].y, rectsPointsA[1].x},
111 {-rectsPointsA[2].y, rectsPointsA[2].x},
112 {-rectsPointsA[3].y, rectsPointsA[3].x},
113 {-rectsPointsB[0].y, rectsPointsB[0].x},
114 {-rectsPointsB[1].y, rectsPointsB[1].x},
115 {-rectsPointsB[2].y, rectsPointsB[2].x},
116 {-rectsPointsB[3].y, rectsPointsB[3].x}
117 };
118
119 rectsPointsA = {
120 a0,a1,a2,a3
121 };
122 rectsPointsB = {
123 b0,b1,b2,b3
124 };
125
126 // Precheck by distance
127 {
128 int insideCount = 0;
129 for (auto& point : rectsPointsA) {
130 float dist = distance(point, rectBCenter);
131 insideCount += dist <= extentBModule ? 1 : 0;
132 }
133 for (auto& point : rectsPointsB) {
134 float dist = distance(point, rectACenter);
135 insideCount += dist <= extentAModule ? 1 : 0;
136 }
137 if (insideCount == 0) {
138 return false;
139 }
140 }
141
142 //Check all axes projections
143 for (auto& axis : testAxes) {
144 // Get proyection range for both rects
145 float maxT_A = -FLT_MAX, minT_A = FLT_MAX;
146 for (auto& point : rectsPointsA) {
147 float t = projectPoint01(point, axis);
148 maxT_A = std::max(maxT_A, t);
149 minT_A = std::min(minT_A, t);
150 }
151
152 float maxT_B = -FLT_MAX, minT_B = FLT_MAX;
153 for (auto& point : rectsPointsB) {
154 float t = projectPoint01(point, axis);
155 maxT_B = std::max(maxT_B, t);
156 minT_B = std::min(minT_B, t);
157 }
158
159 //Check if proyection intersects. When any axis returns no intersection, rects doesn't intersect
160 bool projIntersects =
161 (minT_A <= maxT_B || minT_A >= minT_B) ||
162 (maxT_A <= maxT_B || maxT_A >= minT_B) ||
163 (minT_B <= maxT_A || minT_B >= minT_A) ||
164 (maxT_B <= maxT_A || maxT_B >= minT_A);
165
166 if (!projIntersects) {
167 return false;
168 }
169 }
170 return true;
171}
