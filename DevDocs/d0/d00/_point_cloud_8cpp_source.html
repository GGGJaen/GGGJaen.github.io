121// Calculates a 30-bit Morton code for the

123unsigned int morton3D(float x, float y, float z) {
124 x = std::min(std::max(x * 1024.0f, 0.0f), 1023.0f);
125 y = std::min(std::max(y * 1024.0f, 0.0f), 1023.0f);
