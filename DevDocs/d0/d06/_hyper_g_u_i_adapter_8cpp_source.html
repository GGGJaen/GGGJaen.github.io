330void GEU::HyperGUIAdapter::generateLayerTexture(std::vector<float>& bandBuffer, const int& layer)
331{
332 _activeHypercube->fillCubeBand(layer, bandBuffer);
333
334 std::vector<unsigned char> imageBuffer(bandBuffer.size(), 0);
335 const float maxValue = *std::ranges::max_element(bandBuffer);
336 const float minValue = *std::ranges::min_element(bandBuffer);
337 const float denom = 255 / (maxValue - minValue);
338
339#ifdef _OPENMP
340#pragma omp parallel for
341#endif
342 for (int idx = 0; idx < imageBuffer.size(); ++idx) {
343 imageBuffer[idx] = static_cast<unsigned char>((bandBuffer[idx] - minValue) * denom);
344 }
345
346 const auto layerImage = new GEU::Image(imageBuffer.data(), _activeHypercube->width(), _activeHypercube->height(), 1);
347
348 _layerTexture = new GEU::Texture(layerImage, GL_MIRRORED_REPEAT, GL_MIRRORED_REPEAT, GL_LINEAR_MIPMAP_NEAREST,
349 GL_LINEAR, true, true);
350
351 delete layerImage;
352}

354void GEU::HyperGUIAdapter::assignStepHyperdata()
355{
356 const auto& cloudPoints = _targetPointCloud->getPointsPosition();
357 const float progressInc = 0.85f / (_hyperBandMax - _hyperBandMin);
358 const float preproProgressInc = 0.15f / cloudPoints.size();
359 const auto cloudOffset = _targetPointCloud->getPosition();
360
361 std::vector<int> _insideIdx(cloudPoints.size());
362 int _insideCount = 0;
363 _fusionStatus = LocaleStrings::getInstance()->getString(HIPER_PROGRESS_POINT_CLEANING);
364#ifdef _OPENMP
365#pragma omp parallel for
366#endif
367 for (int i = 0; i < cloudPoints.size(); ++i)
368 {
369 double x, y;
370 auto& point = cloudPoints[i];
371 x = point.x + cloudOffset.x;
372 y = point.y + cloudOffset.y;
373
374 int px_X, px_Y;
375 if (_activeHypercube->geoToPixel(x, y, px_X, px_Y))
376 {
377#ifdef _OPENMP
378#pragma omp critical
379#endif
380 _insideIdx[_insideCount++] = i;
381 }
382
383 _fusionProgress += (preproProgressInc);
384 }
385
386 _fusionProgress = 0.15f;
387 const std::string statusMSG = LocaleStrings::getInstance()->getString(HIPER_PROGRESS_BAND_FUSING);
388 int bandCount = 0;
389 const std::string bands = std::to_string(_hyperBandMax - _hyperBandMin);
390 _fusionStatus = statusMSG + " (" + std::to_string(bandCount) + " / " + bands + ")";
391#ifdef _OPENMP
392#pragma omp parallel for
393#endif
394 for (int b = _hyperBandMin; b < _hyperBandMax; ++b)
395 {
396 std::vector<float> rasterBand, cloudBandValues(cloudPoints.size(), 0);
397 _activeHypercube->fillCubeBand(b, rasterBand);
398 double x, y;
399
400 for (size_t idx = 0; idx < _insideIdx.size(); ++idx)
401 {
402 const auto i = _insideIdx[idx];
403
404 auto& point = cloudPoints[i];
405 x = point.x + cloudOffset.x;
406 y = point.y + cloudOffset.y;
407
408 int px_X, px_Y;
409 if (_activeHypercube->geoToPixel(x, y, px_X, px_Y))
410 {
411 cloudBandValues[i] = rasterBand[_activeHypercube->flattenIndex(px_Y, px_X, 0)];
412 }
413 }
414 _targetPointCloud->setFusionData(PointCloud::dataSource::HYPERSPECTRAL, b, cloudBandValues);
415
416
417#ifdef _OPENMP
418#pragma omp critical
419#endif
420 {
421 bandCount += 1;
422 _fusionProgress += (progressInc);
423 _fusionStatus = statusMSG + " (" + std::to_string(bandCount) + " / " + bands + ")";
424 }
425 }
426 _fusionStatus = "";
427}
