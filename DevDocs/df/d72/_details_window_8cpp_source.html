18void GEU::DetailsWindow::renderWindow() {
19 if (ImGui::Begin(getIdTitle().c_str(), &_windowOpen, _windowFlags)) {
20 if (InspectorWindow::_selectedInspectorItem < 0) {
21 GuiUtilities::Text(DETAILS_NOSELECTION);
22 }
23 else {
24 static Renderer* _renderer = Renderer::getInstance();
25
26
27 switch (InspectorWindow::_selectedItemType) {
28 case InspectorWindow::SceneItemType::Camera:
29 _renderer->getActiveScene()->dirty |= makeCameraItemInspector();
30 break;
31 case InspectorWindow::SceneItemType::Light:
32 _renderer->getActiveScene()->dirty |= makeLightItemInspector();
33 break;
34 case InspectorWindow::SceneItemType::Material:
35 makeMaterialItemInspector();
36 break;
37 case InspectorWindow::SceneItemType::PCloud:
38 _renderer->getActiveScene()->dirty |= makePCloudItemInspector();
39 break;
40 case InspectorWindow::SceneItemType::Model3D:
41 _renderer->getActiveScene()->dirty |= makeModelItemInspector();
42 break;
43 case InspectorWindow::SceneItemType::Image:
44 _renderer->getActiveScene()->dirty |= makeImageItemInspector();
45 break;
46 default:
47 GuiUtilities::Text(DETAILS_NOSELECTION);
48 break;
49 }
50 }
51 }
52 ImGui::End();
53}
