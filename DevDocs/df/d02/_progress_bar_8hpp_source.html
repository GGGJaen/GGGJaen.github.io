123inline void ProgressBar::update() {
124
125 if (n_cycles == 0) throw std::runtime_error(
126 "progressbar::update: number of cycles not set");
127
128 if (!update_is_called) {
129 if (do_show_bar == true) {
130 std::cout << opening_bracket_char;
131 for (int _ = 0; _ < 50; _++) std::cout << todo_char;
132 std::cout << closing_bracket_char << " 0%";
133 }
134 else std::cout << "0%";
135 }
136 update_is_called = true;
137
138 int perc = 0;
139
140 // compute percentage, if did not change, do nothing and return
141 perc = progress*100./(n_cycles-1);
142 if (perc < last_perc) return;
143
144 // update percentage each unit
145 if (perc == last_perc + 1) {
146 // erase the correct number of characters
147 if (perc <= 10) std::cout << "\b\b" << perc << '%';
148 else if (perc > 10 and perc < 100) std::cout << "\b\b\b" << perc << '%';
149 else if (perc == 100) std::cout << "\b\b\b" << perc << '%';
150 }
151 if (do_show_bar == true) {
152 // update bar every ten units
153 if (perc % 2 == 0) {
154 // erase closing bracket
155 std::cout << std::string(closing_bracket_char.size(), '\b');
156 // erase trailing percentage characters
157 if (perc < 10) std::cout << "\b\b\b";
158 else if (perc >= 10 && perc < 100) std::cout << "\b\b\b\b";
159 else if (perc == 100) std::cout << "\b\b\b\b\b";
160
161 // erase 'todo_char'
162 for (int j = 0; j < 50-(perc-1)/2; ++j) {
163 std::cout << std::string(todo_char.size(), '\b');
164 }
165
166 // add one additional 'done_char'
167 if (perc == 0) std::cout << todo_char;
168 else std::cout << done_char;
169
170 // refill with 'todo_char'
171 for (int j = 0; j < 50-(perc-1)/2-1; ++j) std::cout << todo_char;
172
173 // readd trailing percentage characters
174 std::cout << closing_bracket_char << ' ' << perc << '%';
175 }
176 }
177 ++progress;
178 last_perc = perc;
179 std::cout << std::flush;
180
181 return;
182}
