26 startQ1 = 0;
27 endQ3 = sortedTemperature.size() - 1;
28
29 // Compute each quartile by splitting the vector
30 if (sortedTemperature.size() % 2) // Odd
31 {
32 medianQ2Index = std::floor(sortedTemperature.size() / 2.0f);
33 endQ1 = startQ3 = medianQ2Index;
34
35 _q1 = sortedTemperature[(endQ1 + startQ1) / 2];
36 _q3 = sortedTemperature[(endQ3 + startQ3) / 2];
37 } else // Even
38 {
39 endQ1 = sortedTemperature.size() / 2.0f - 1;
40 startQ3 = endQ1 + 1;
41
42 _q1 = (sortedTemperature[std::floor((endQ1 + startQ1) / 2.0f)] + sortedTemperature[std::ceil((endQ1 + startQ1) / 2.0f)]) / 2.0f;
43 _q3 = (sortedTemperature[std::floor((endQ3 + startQ3) / 2.0f)] + sortedTemperature[std::ceil((endQ3 + startQ3) / 2.0f)]) / 2.0f;
44 }
45
46 _interquartileRange = _q3 - _q1;
