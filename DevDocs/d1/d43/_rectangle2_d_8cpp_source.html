46bool Rectangle2D::isInsideBinarySearch(const vec2& point) {
47 // Do binary search over polygon vertices to find the fan triangle (v[0], v[low], v[high])
48 int low = 0, high = NUM_RECTANGLE_CORNERS;
49
50 do {
51 int mid = (low + high) / 2;
52
53 if (triangleIsCCW(_corners[0], _corners[mid], point)) {
54 low = mid;
55 } else {
56 high = mid;
57 }
58
59 } while (low + 1 < high);
60
61 // If point outside last (or first) edge, then it is not inside the n-gon
62 if (low == 0 || high == NUM_RECTANGLE_CORNERS) return 0;
63
64 // point is inside the polygon if it is left of the directed edge from v[low] to v[high]
65 return triangleIsCCW(_corners[low], _corners[high], point);
66}
