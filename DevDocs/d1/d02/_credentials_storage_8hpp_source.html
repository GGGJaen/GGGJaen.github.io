142 inline json::jobject CredentialsStorage::GetFileAsJSON()

147 throw std::runtime_error("Credentials file not found!");

159 inline json::jobject CredentialsStorage::GetServiceObject(Service service)
160 {
161 if (!ServiceKeys.contains(service))

163 throw std::runtime_error("Service credentials unavailable");
164 }
165
166 auto credentialsContents = GetFileAsJSON();
