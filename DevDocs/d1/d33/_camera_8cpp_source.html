171void GEU::Camera::orbitXZ(double speed) {
172 const mat4 rotationMatrix = glm::rotate(dmat4(1.0f), speed, this->_properties._u); // We will pass over the scene, x or z axis could be used
173
174
175 this->_properties._eye = dvec3(rotationMatrix * dvec4(this->_properties._eye - this->_properties._lookAt, 1.0f)) + this->_properties._lookAt;
176 this->_properties._u = dvec3(rotationMatrix * vec4(this->_properties._u, 0.0f));
177 this->_properties._v = dvec3(rotationMatrix * vec4(this->_properties._v, 0.0f));
178 this->_properties._n = dvec3(rotationMatrix * vec4(this->_properties._n, 0.0f));
179 this->_properties._up = glm::normalize(glm::cross(this->_properties._n, this->_properties._u)); // Free rotation => we can look down or up
180
181 if (trackedModel)
182 setLookAt(trackedModel->getAABB().center());
183
184 this->_properties.computeViewMatrices();
185}

202void GEU::Camera::pan(double speed) {
203 const dmat4 rotationMatrix = glm::rotate(dmat4(1.0f), speed, this->_properties._v);
204
205 // Up vector can change, not in the original position tho. Example: orbit XZ (rotated camera) + pan
206 this->_properties._u = dvec3(rotationMatrix * vec4(this->_properties._u, 0.0f));
207 this->_properties._v = dvec3(rotationMatrix * vec4(this->_properties._v, 0.0f));
208 this->_properties._n = dvec3(rotationMatrix * vec4(this->_properties._n, 0.0f));
209 this->_properties._up = glm::normalize(glm::cross(this->_properties._n, this->_properties._u));
210 this->_properties._lookAt = dvec3(rotationMatrix * vec4(this->_properties._lookAt - this->_properties._eye, 1.0f)) + this->_properties._eye;
211
212 this->_properties.computeViewMatrices();
213}

215void GEU::Camera::tilt(double speed) {
216 const mat4 rotationMatrix = glm::rotate(dmat4(1.0f), speed, this->_properties._u);
217
218 const auto n = glm::vec3(rotationMatrix * glm::dvec4(this->_properties._n, 0.0f));
219 const float alpha = glm::acos(glm::dot(n, glm::vec3(0.0f, 1.0f, 0.0f)));
220
221 if (alpha < speed || alpha >(glm::pi<float>() - speed)) {
222 return;
223 }
224
225 this->_properties._v = glm::dvec3(rotationMatrix * glm::dvec4(this->_properties._v, 0.0f));
226 this->_properties._n = n;
227 this->_properties._up = glm::normalize(glm::cross(this->_properties._n, this->_properties._u)); // It could change because of the rotation
228 this->_properties._lookAt = dvec3(rotationMatrix * glm::dvec4(this->_properties._lookAt - this->_properties._eye, 1.0f)) + this->_properties._eye;
229
230 this->_properties.computeViewMatrices();
231}

254vec2 GEU::Camera::projectPoint(vec4 point) const
255{
256 vec4 projPoint = _properties._viewProjectionMatrix * point;
257 projPoint /= projPoint.w;
