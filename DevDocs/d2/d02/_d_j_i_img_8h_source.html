23inline u16 DJIImg::vignettingCorrectionPerPixel(const u16 pVal, const vec2 pixel) const {
24 //https://dl.djicdn.com/downloads/DJI_Mavic_3_Enterprise/20230829/Mavic_3M_Image_Processing_Guide_EN.pdf
25 const float r = std::sqrtf(powf(pixel.x - vignetteCx, 2) + powf(pixel.y - vignetteCy, 2));
26 const float r2 = r * r;
27 const float r3 = r2 * r;
28 const float r4 = r3 * r;
29 const float r5 = r4 * r;
30 const float r6 = r5 * r;

32 const float vignetteCalibration = 1 + vignettePolynomial[0] * r + vignettePolynomial[1] * r2 +
33 vignettePolynomial[2] * r3 + vignettePolynomial[3] * r4 +
34 vignettePolynomial[4] * r5 + vignettePolynomial[5] * r6;
35
36 return pVal * vignetteCalibration;
