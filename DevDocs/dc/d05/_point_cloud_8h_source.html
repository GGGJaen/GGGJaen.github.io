267 PointCloud::u32Bits PointCloud::getAttribute(const u32 pointIndex, const u16 band) {
268 _controlledByRenderer = false;
269 switchAttribute(Att, band);
270 u32 meshletIndex = pointIndex / _meshletSize;
271 u16 pointInsideMeshlet = pointIndex % _meshletSize;
272 requestMeshletData(meshletIndex, _meshletSize);
273 while (_meshlets[meshletIndex].pointsLoaded <= pointInsideMeshlet) {
274 std::this_thread::yield();
