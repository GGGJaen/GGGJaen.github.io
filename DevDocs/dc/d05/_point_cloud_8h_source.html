233 PointCloud::u32Bits PointCloud::getAttribute(const u32 pointIndex, const u16 band) {
234 _controlledByRenderer = false;
235 switchAttribute(Att, band);
236 u32 meshletIndex = pointIndex / _meshletSize;
237 u16 pointInsideMeshlet = pointIndex % _meshletSize;
238 requestMeshletData(meshletIndex, _meshletSize);
239 while (_meshlets[meshletIndex].pointsLoaded <= pointInsideMeshlet) {
240 std::this_thread::yield();
241 }
242 return _meshlets[meshletIndex].att[pointInsideMeshlet];
243 }
