50bool GEU::TreeSegmentationGUIAdapter::makeImageWithRects(ImVec2& rectBegin, ImVec2& rectSize)
51{
52 const auto& segResult = _imageSegmentation._segmentationResult;
53 constexpr float zoomSize = 48.0f;
54 constexpr float zoomFactor = 8.0f;
55
56 const auto imageInitPos = ImGui::GetCursorPos(), imageScreenPos = ImGui::GetCursorScreenPos();
57 const auto tooltipArea = GuiUtilities::makeImageZoomable(_imageSegmentation._inputImageTexture, ImGui::GetStyle().WindowPadding, false, zoomSize, zoomFactor, true);
58 const ImVec2 imageRectSize = ImGui::GetItemRectSize();
59
60 const auto drawList = ImGui::GetWindowDrawList();
61 const auto rectDrawColor = IM_COL32(_segmentRectColor[0] * 255, _segmentRectColor[1] * 255, _segmentRectColor[2] * 255, 255);
62 const auto rectDrawFillColor = IM_COL32(_segmentRectColor[0] * 255, _segmentRectColor[1] * 255, _segmentRectColor[2] * 255, 75);
63
64 if (_previewSegmentSizeRanges)
65 {
66 drawList->AddRectFilled(imageScreenPos, imageScreenPos + imageRectSize, IM_COL32(0, 0, 0, 128));
67 drawList->AddRect(imageScreenPos, imageScreenPos + (imageRectSize * _imageSegmentation._segmentSizeLimits.y), IM_COL32(0, 0, 0, 255));
68 drawList->AddRect(imageScreenPos, imageScreenPos + (imageRectSize * _imageSegmentation._segmentSizeLimits.x), IM_COL32(255, 255, 255, 255));
69 }
70 else
71 {
72
73 for (const auto& segmentRect : segResult)
74 {
75 ImVec2 rectMin(
76 imageScreenPos.x + segmentRect._min.x * imageRectSize.x,
77 imageScreenPos.y + segmentRect._min.y * imageRectSize.y);
78
79 ImVec2 rectMax(
80 imageScreenPos.x + segmentRect._max.x * imageRectSize.x,
81 imageScreenPos.y + segmentRect._max.y * imageRectSize.y);
82
83 drawList->AddRect(rectMin, rectMax, rectDrawColor);
84 drawList->AddRectFilled(rectMin, rectMax, rectDrawFillColor);
85
86 if (tooltipArea._areaDrawList &&
87 !(segmentRect._max.x < tooltipArea._uvBegin.x || segmentRect._min.x > tooltipArea._uvEnd.x ||
88 segmentRect._max.y < tooltipArea._uvBegin.y || segmentRect._min.y > tooltipArea._uvEnd.y))
89 {
90 const auto zoomUVSize = tooltipArea.getUVSize();
91 const auto zoomAreaSize = tooltipArea.getAreaSize();
92 const ImVec2 zoomedMin(
93 std::clamp((segmentRect._min.x - tooltipArea._uvBegin.x) / zoomUVSize.x, 0.0f, 1.0f),
94 std::clamp((segmentRect._min.y - tooltipArea._uvBegin.y) / zoomUVSize.y, 0.0f, 1.0f)
95 );
96 const ImVec2 zoomedMax(
97 std::clamp((segmentRect._max.x - tooltipArea._uvBegin.x) / zoomUVSize.x, 0.0f, 1.0f),
98 std::clamp((segmentRect._max.y - tooltipArea._uvBegin.y) / zoomUVSize.y, 0.0f, 1.0f)
99 );
100
101 rectMin = {
102 tooltipArea._areaBegin.x + zoomedMin.x * zoomAreaSize.x,
103 tooltipArea._areaBegin.y + zoomedMin.y * zoomAreaSize.y
104 };
105 rectMax = {
106 tooltipArea._areaBegin.x + zoomedMax.x * zoomAreaSize.x,
107 tooltipArea._areaBegin.y + zoomedMax.y * zoomAreaSize.y
108 };
109
110
111 tooltipArea._areaDrawList->AddRect(rectMin, rectMax, rectDrawColor);
112 tooltipArea._areaDrawList->AddRectFilled(rectMin, rectMax, rectDrawFillColor);
113
114 }
115 }
116
117 }
118 rectBegin = imageScreenPos;
119 rectSize = imageRectSize;
120
121 ImGui::SetCursorPos(imageInitPos);
122 return ImGui::InvisibleButton("##treeseg_imageinvbtn", imageRectSize);
123}
