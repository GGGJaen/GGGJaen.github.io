25inline float MicasenseImg::RadiometricCalibration(const float p_value, const vec2 pixel) const {
26 //https://support.micasense.com/hc/en-us/articles/115000351194-Radiometric-Calibration-Model-for-MicaSense-Sensors
27 const float r = std::sqrtf(powf(pixel.x - vignetteCx, 2) + pow(pixel.y - vignetteCy, 2));
28 const float vignetteCalibration = 1 + vignettePolynomial[0] * r + vignettePolynomial[1] * powf(r, 2) +
29 vignettePolynomial[2] * powf(r, 3) + vignettePolynomial[3] * powf(r, 4) +
30 vignettePolynomial[4] * powf(r, 5) + vignettePolynomial[5] * powf(r, 6);
31
32 const float intensity = radiometricCalibration[0] / gain * (p_value - blackLevel) / (exposureTime + radiometricCalibration[1] * pixel.y - radiometricCalibration[2] * exposureTime * pixel.y);
33 if (intensity < 0)
34 return 0.0f;
35
36 return intensity / vignetteCalibration;
37}
