578 queryDatoCrudo += " AND DISTINTIVO=\'" + distintivo + "\'";
579 }

584 }

587 std::string queryDatoCrudo = query + " WHERE VUELOS_FECHA IN (";

599 queryDatoCrudo += std::to_string(it.codMunicipio) + ", ";

603

605 queryDatoCrudo += " AND TIPO IN (" + tipo + ")";
