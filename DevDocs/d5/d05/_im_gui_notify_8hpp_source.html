302 ImGuiToastPhase getPhase() {
303 const int64_t elapsed = std::chrono::duration_cast<std::chrono::milliseconds>(getElapsedTime()).count();
304
305 if (this->progressPerc != nullptr) {
306 if (elapsed < NOTIFY_FADE_IN_OUT_TIME) {
307 return ImGuiToastPhase::FadeIn;
308 }
309 return *this->progressPerc >= 1.0f ? ImGuiToastPhase::Expired : ImGuiToastPhase::Wait;
310 }
311
312 if (elapsed > NOTIFY_FADE_IN_OUT_TIME + this->dismissTime + NOTIFY_FADE_IN_OUT_TIME) {
313 return ImGuiToastPhase::Expired;
314 } else
315 if (elapsed > NOTIFY_FADE_IN_OUT_TIME + this->dismissTime) {
316 return ImGuiToastPhase::FadeOut;
317 } else
318 if (elapsed > NOTIFY_FADE_IN_OUT_TIME) {
319 return ImGuiToastPhase::Wait;
320 } else {
321 return ImGuiToastPhase::FadeIn;
322 }
323 }
324
329 float getFadePercent() {
330 const ImGuiToastPhase phase = getPhase();
331 const int64_t elapsed = std::chrono::duration_cast<std::chrono::milliseconds>(getElapsedTime()).count();
332
333 if (phase == ImGuiToastPhase::FadeIn) {
334 return ((float)elapsed / (float)NOTIFY_FADE_IN_OUT_TIME) * NOTIFY_OPACITY;
335 } else
336 if (phase == ImGuiToastPhase::FadeOut) {
337 return (1.f - (((float)elapsed - (float)NOTIFY_FADE_IN_OUT_TIME - (float)this->dismissTime) / (float)NOTIFY_FADE_IN_OUT_TIME)) * NOTIFY_OPACITY;
338 }
339
340 return 1.f * NOTIFY_OPACITY;
341 }

461 inline void RenderNotifications() {
462 const ImVec2 mainWindowSize = GetMainViewport()->Size;
463
464 float height = 0.f;
465
466 for (size_t i = 0; i < notifications.size(); i++) {
467 ImGuiToast* currentToast = &notifications[i];
468
469 // Remove toast if expired
470 if (currentToast->getPhase() == ImGuiToastPhase::Expired) {
471 RemoveNotification(i);
472 continue;
473 }
474
475 // Get icon, title and other data
476 const char* icon = currentToast->getIcon();
477 const char* title = currentToast->getTitle();
478 const char* content = currentToast->getContent();
479 std::string defaultTitle = currentToast->getDefaultTitle();
480 const float opacity = currentToast->getFadePercent(); // Get opacity based of the current phase
481
482 // Window rendering
483 ImVec4 textColor = currentToast->getColor();
484 textColor.w = opacity;
485
486 // Generate new unique name for this toast
487 char windowName[50];
488 #ifdef _WIN32
489 sprintf_s(windowName, "##TOAST%d", (int)i);
490 #elif defined(__linux__)
491 sprintf(windowName, "##TOAST%d", (int)i);
492 #else
493 throw "Unsupported platform";
494 #endif
495
496 //PushStyleColor(ImGuiCol_Text, textColor);
497 SetNextWindowBgAlpha(opacity);
498
499 // Set notification window position to bottom right corner of the main window, considering the main window size and location in relation to the display
500 ImVec2 mainWindowPos = GetMainViewport()->Pos;
501 SetNextWindowPos(ImVec2(mainWindowPos.x + mainWindowSize.x - NOTIFY_PADDING_X, mainWindowPos.y + mainWindowSize.y - NOTIFY_PADDING_Y - height), ImGuiCond_Always, ImVec2(1.0f, 1.0f));
502
503 // Set notification window flags
504 if (!NOTIFY_USE_DISMISS_BUTTON && currentToast->getOnButtonPress() == nullptr) {
505 currentToast->setWindowFlags(NOTIFY_DEFAULT_TOAST_FLAGS | ImGuiWindowFlags_NoInputs);
506 }
507
508 Begin(windowName, nullptr, currentToast->getWindowFlags());
509
510 // Render over all other windows
511 BringWindowToDisplayFront(GetCurrentWindow());
512
513 // Here we render the toast content
514 {
515 PushTextWrapPos(mainWindowSize.x / 3.f); // We want to support multi-line text, this will wrap the text after 1/3 of the screen width
516
517 bool wasTitleRendered = false;
518
519 // If an icon is set
520 if (!NOTIFY_NULL_OR_EMPTY(icon)) {
521 //Text(icon); // Render icon text
522 TextColored(textColor, "%s", icon);
523 wasTitleRendered = true;
524 }
525
526 // If a title is set
527 if (!NOTIFY_NULL_OR_EMPTY(title)) {
528 // If a title and an icon is set, we want to render on same line
529 if (!NOTIFY_NULL_OR_EMPTY(icon))
530 SameLine();
531
532 Text("%s", title); // Render title text
533 wasTitleRendered = true;
534 } else
535 if (!defaultTitle.empty()) {
536 if (!NOTIFY_NULL_OR_EMPTY(icon))
537 SameLine();
538
539 Text("%s", defaultTitle.c_str()); // Render default title text (ImGuiToastType_Success -> "Success", etc...)
540 wasTitleRendered = true;
541 }
542
543 // If a dismiss button is enabled
544 if (NOTIFY_USE_DISMISS_BUTTON) {
545 // If a title or content is set, we want to render the button on the same line
546 if (wasTitleRendered || !NOTIFY_NULL_OR_EMPTY(content)) {
547 SameLine();
548 }
549
550 // Render the dismiss button on the top right corner
551 // NEEDS TO BE REWORKED
552 float scale = 0.8f;
553
554 if (CalcTextSize(content).x > GetWindowContentRegionMax().x) {
555 scale = 0.95f;
556 }
557
558 SetCursorPosX(GetCursorPosX() + (GetWindowSize().x - GetCursorPosX()) * scale);
559
560 // If the button is pressed, we want to remove the notification
561 if (Button(ICON_FA_XMARK)) {
562 RemoveNotification(i);
563 }
564 }
565
566 // In case ANYTHING was rendered in the top, we want to add a small padding so the text (or icon) looks centered vertically
567 if (wasTitleRendered && !NOTIFY_NULL_OR_EMPTY(content)) {
568 SetCursorPosY(GetCursorPosY() + 5.f); // Must be a better way to do this!!!!
569 }
570
571 // If a content is set
572 if (!NOTIFY_NULL_OR_EMPTY(content)) {
573 if (wasTitleRendered) {
574 #if NOTIFY_USE_SEPARATOR
575 Separator();
576 #endif
577 }
578 if (currentToast->getProgressPerc() != nullptr) {
579 Text("%s %s", content, (static_cast<int>(ImGui::GetTime() / 0.1f) & 4) < 2 ? ICON_FA_CIRCLE : ICON_FA_CIRCLE_DOT); // Render content text
580 }
581 else {
582 Text("%s", content); // Render content text
583 }
584 }
585
586 // If a progress bar float is set
587 if (currentToast->getProgressPerc() != nullptr) {
588
589 float perc = *currentToast->getProgressPerc();
590
591 ProgressBar(perc, ImVec2(NOTIFY_PROGRESS_WIDTH, 0));
592
593 }
594
595 // If a button is set
596 if (currentToast->getOnButtonPress() != nullptr) {
597 // If the button is pressed, we want to execute the lambda function
598 if (Button(currentToast->getButtonLabel())) {
599 currentToast->getOnButtonPress()();
600 }
601 }
602
603 PopTextWrapPos();
604 }
605
606 // Save height for next toasts
607 height += GetWindowHeight() + NOTIFY_PADDING_MESSAGE_Y;
608
609 // End
610 End();
611 }
612 }
