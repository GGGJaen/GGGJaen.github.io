66 void reserveGPUMemory(const SSBOSlots ssboSlot, const size_t dataCount, const GLbitfield memFlags = GL_DYNAMIC_STORAGE_BIT) {
67 const size_t bufferSize = dataCount * sizeof(T);
68 if (bufferSize <= maxMemoryPerSSBO) {
69 if (_ssboIds[(uint8_t)ssboSlot] != 0) {
70 clearResource(ssboSlot);
71 }
72 glCreateBuffers(1, &_ssboIds[(uint8_t)ssboSlot]);
73 glNamedBufferStorage(_ssboIds[(uint8_t)ssboSlot], bufferSize, nullptr, memFlags);
74 _memoryReserved[(uint8_t)ssboSlot] = bufferSize;
75 totalUsedMemory += _memoryReserved[(uint8_t)ssboSlot];
76 } else {
77 spdlog::warn("Trying to reserve more memory than permited by MAX_SHADER_STORAGE_BLOCK_SIZE");
78 }
79 }
