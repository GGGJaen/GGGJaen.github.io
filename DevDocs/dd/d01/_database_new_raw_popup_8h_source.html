180inline double gpsDistance(double lat1, double long1,
181 double lat2, double long2) {
182 // Convert the latitudes
183 // and longitudes
184 // from degree to radians.
185 lat1 = toRadians(lat1);
186 long1 = toRadians(long1);
187 lat2 = toRadians(lat2);
188 long2 = toRadians(long2);
189
190 // Haversine Formula
191 const double dlong = long2 - long1;
192 const double dlat = lat2 - lat1;
193
194 double ans = pow(sin(dlat / 2), 2) +
195 cos(lat1) * cos(lat2) *
196 pow(sin(dlong / 2), 2);
197
198 ans = 2 * asin(sqrt(ans));
199
200 // Radius of Earth in
201 // Kilometers, R = 6371
202 constexpr double R = 6371;
203
204 // Calculate the result
205 ans = ans * R;
206
207 return ans;
208}
