63 void waitForLockedRange(const size_t lockBeginBytes, const size_t lockLength, const bool waitOnCpu) {
64 const BufferRange testRange = { lockBeginBytes, lockLength };
65 std::vector<BufferLock> swapLocks;
66 for (auto& bufferLock : bufferLocks) {

94 static void wait(const GLsync* syncObj, const bool waitOnCPU) {
95 if (waitOnCPU) {
96 GLbitfield waitFlags = 0;
97 GLuint64 waitDuration = 0;
98 bool keepGoing = true;
99 while (keepGoing) {
100 const GLenum waitRet = glClientWaitSync(*syncObj, waitFlags, waitDuration);
101
102 if (waitRet == GL_WAIT_FAILED) {
103 spdlog::warn("Buffer lock wait failed for some reason");
104 return;
105 }
106
107 keepGoing = !(waitRet == GL_ALREADY_SIGNALED || waitRet == GL_CONDITION_SATISFIED);
108
109 // After the first time, need to start flushing, and wait for a looong time.
110 waitFlags = GL_SYNC_FLUSH_COMMANDS_BIT;
111 waitDuration = 100000000;
112 }
113 } else {
114 glWaitSync(*syncObj, 0, GL_TIMEOUT_IGNORED);
115 }
116 }
117
118 static void cleanup(const BufferLock* bufferLock) {
119 glDeleteSync(bufferLock->syncObj);
120 }

193 const size_t bufferSize = dataCount * sizeof(T);

196 clearResource(ssboSlot);

199 glNamedBufferStorage(_ssboIds[(uint8_t)ssboSlot], bufferSize, nullptr, memFlags);
