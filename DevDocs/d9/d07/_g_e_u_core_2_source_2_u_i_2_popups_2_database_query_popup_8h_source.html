567 if (!fecha.empty()) {

570 queryVuelo += " AND ZONA_UBICACION=\'" + zona->ubicacion + "\' AND ZONA_MUNICIPIO_CODIGO=" + std::to_string(zona->codMunicipio);

576 } else {

579 queryDatoCrudo += " WHERE TIPO IN (" + tipo + ")";
580 if (!distintivo.empty()) {
581 queryDatoCrudo += " AND DISTINTIVO=\'" + distintivo + "\'";
582 }

587 }
588

590 std::string queryDatoCrudo = query + " WHERE VUELOS_FECHA IN (";
591 for (auto& it : vueloResult) {
592 queryDatoCrudo += "TO_DATE(\'" + it.fecha.toText("YYYY-MM-DD") + "\', \'YYYY-MM-DD\'),";

594 queryDatoCrudo.pop_back();

602 queryDatoCrudo += std::to_string(it.codMunicipio) + ", ";
