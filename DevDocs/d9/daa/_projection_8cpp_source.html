279vec2 fisheyeProjectPoint(const PointXYZ& transformedPoint, const Vec4d& A, const Vec4d& D, const float& cx, const float& cy) {
280 constexpr float twoDividedByPI = 2.0f / M_PI;
281 const float r = sqrtf(transformedPoint.x * transformedPoint.x + transformedPoint.y * transformedPoint.y);
282 const float theta = twoDividedByPI * atan(r / transformedPoint.z);
283 const float poly = D[0] + D[1] * theta + D[2] * theta * theta + D[3] * theta * theta * theta;
284 const float xh = (poly * transformedPoint.x) / r;
285 const float yh = (poly * transformedPoint.y) / r;
286
287 //Coordenadas de la imagen
288 const int px = A[0] * xh + A[1] * yh + cx;
289 const int py = A[2] * xh + A[3] * yh + cy;
290 return { px, py };
291}
