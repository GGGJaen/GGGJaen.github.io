367 const auto ordinates = json::jobject::parse(jsonObject.get("SDO_ORDINATES"));
368 for (size_t o = 0; o < ordinates.size(); ++o) {
369 g._ordinate_array.push_back(ordinates.array(o));
370 }
371 if (g._ordinate_array.empty() && !jsonObject["SDO_POINT"].is_null()) {
372 const auto point = json::jobject::parse(jsonObject.get("SDO_POINT"));
373 g._ordinate_array.push_back(point["X"]);
374 g._ordinate_array.push_back(point["Y"]);
