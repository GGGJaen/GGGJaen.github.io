56bool AABB::rayIntersects(const vec3& origin, const vec3& direction) const
57{ // https://web.archive.org/web/20090803054252/http://tog.acm.org/resources/GraphicsGems/gems/RayBox.c
58 static constexpr unsigned NUMDIM = 3;
59 static constexpr unsigned RIGHT = 0;
60 static constexpr unsigned LEFT = 1;
61 static constexpr unsigned MIDDLE = 2;
62
63 double coord[NUMDIM]; /*ray */
64 {
65 char inside = TRUE;
66 char quadrant[NUMDIM];
67 int i;
68 double maxT[NUMDIM];
69 double candidatePlane[NUMDIM];
70
71 /* Find candidate planes; this loop can be avoided if
72 rays cast all from the eye(assume perpsective view) */
73 for (i = 0; i < NUMDIM; i++)
74 if (origin[i] < _min[i]) {
75 quadrant[i] = LEFT;
76 candidatePlane[i] = _min[i];
77 inside = FALSE;
78 }
79 else if (origin[i] > _max[i]) {
80 quadrant[i] = RIGHT;
81 candidatePlane[i] = _max[i];
82 inside = FALSE;
83 }
84 else {
85 quadrant[i] = MIDDLE;
86 }
87
88 /* Ray origin inside bounding box */
89 if (inside) {
90 return true;
91 }
92
93
94 /* Calculate T distances to candidate planes */
95 for (i = 0; i < NUMDIM; i++)
96 if (quadrant[i] != MIDDLE && direction[i] != 0.)
97 maxT[i] = (candidatePlane[i] - origin[i]) / direction[i];
98 else
99 maxT[i] = -1.;
100
101 /* Get largest of the maxT's for final choice of intersection */
102 int whichPlane = 0;
103 for (i = 1; i < NUMDIM; i++)
104 if (maxT[whichPlane] < maxT[i])
105 whichPlane = i;
106
107 /* Check final candidate actually inside box */
108 if (maxT[whichPlane] < 0.) return false;
109 for (i = 0; i < NUMDIM; i++)
110 if (whichPlane != i) {
111 coord[i] = origin[i] + maxT[whichPlane] * direction[i];
112 if (coord[i] < _min[i] || coord[i] > _max[i])
113 return false;
114 }
115 else {
116 coord[i] = candidatePlane[i];
117 }
118 return true; /* ray hits box */
119 }
