158GEU::TriangleMesh::Component* GEU::TriangleMesh::processMesh(aiMesh* mesh, const aiScene* scene, const std::string& folder) {
159 std::vector<VAO::Vertex> vertices(mesh->mNumVertices);
160 std::vector<GLuint> indices(mesh->mNumFaces);
161 AABB aabb;
162 Material* material = nullptr;
163 Material::MaterialDescription description;
164
165 // Vertices
166 #pragma omp parallel for
167 for (int i = 0; i < mesh->mNumVertices; i++) {
168 VAO::Vertex vertex;
169 vertex._position = vec3(mesh->mVertices[i].x, mesh->mVertices[i].y, mesh->mVertices[i].z);
170 vertex._normal = vec3(mesh->mNormals[i].x, mesh->mNormals[i].y, mesh->mNormals[i].z);
171 if (mesh->mTextureCoords[0]) {
172 vertex._textCoord = vec2(mesh->mTextureCoords[0][i].x, mesh->mTextureCoords[0][i].y);
173 } else {
174 vertex._textCoord = vec2(.0f);
175 }
176
177 vertices[i] = vertex;
178
179 #pragma omp critical
180 aabb.update(vertex._position);
181 }
182
183 // Indices
184 for (unsigned int i = 0; i < mesh->mNumFaces; i++) {
185 aiFace face = mesh->mFaces[i];
186 for (unsigned int j = 0; j < face.mNumIndices; j++)
187 indices.push_back(face.mIndices[j]);
188
189 indices[i] = RESTART_PRIMITIVE_INDEX;
190 }
191
192 // Material
193 if (mesh->mMaterialIndex >= 0) {
194 aiMaterial* aiMaterial = nullptr;
195 aiMaterial = scene->mMaterials[mesh->mMaterialIndex];
196 std::string materialName = aiMaterial->GetName().C_Str();
197 MaterialList* materialList = MaterialList::getInstance();
198
199 description = std::move(Material::getMaterialDescription(aiMaterial, folder));
200
201 if (!(material = materialList->isMaterialAvailable(materialName))) {
202 this->_material = material = loadMaterial(&description, materialName);
203 materialList->saveMaterial(materialName, material);
204 }
205 }
206
207 auto* component = new Component;
208 component->_vertices = std::move(vertices);
209 component->_indices = std::move(indices);
210 component->_aabb = aabb;
211 component->_material = material;
212 component->_materialDescription = description;
213 component->_baseIndex = _numIndices;
214 component->_baseVertex = _numVertices;
215 component->_name = std::string(mesh->mName.C_Str());
216
217 _numIndices += component->_indices.size();
218 _numVertices += component->_vertices.size();
219
220 // Bones
221 for (unsigned int i = 0; i < mesh->mNumBones; i++) {
222 aiBone* bone = mesh->mBones[i];
223
224 for (unsigned int i = 0; i < bone->mNumWeights; i++) {
225 const aiVertexWeight& vw = bone->mWeights[i];
226 size_t vertexID = bone->mWeights[i].mVertexId;
227 }
228 }
