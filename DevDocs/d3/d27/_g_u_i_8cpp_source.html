185void GEU::GUI::handleNextPopup() {
186 if (!_pendingPopups.empty() && !_currentPopup) {
187 _currentPopup = _pendingPopups.front();
188 _pendingPopups.pop_front();
189
190 _currentPopup->openPopup();
191 }
192
193 if (_currentPopup) {
194 //ImGui::SetNextWindowSizeConstraints(MIN_POPUP_SIZE, MAX_POPUP_SIZE);
195 const bool popupClosing = _currentPopup->renderPopup(InputManager::getApplicationState(), _renderer->getActiveScene());
196 if (popupClosing or !_currentPopup->isOpen()) {
197 delete _currentPopup;
198 _currentPopup = nullptr;
199 }

204void GEU::GUI::makeMainDockNode() {
205 const ImGuiViewport* imguiViewport = ImGui::GetMainViewport();
206
207 //https://gist.github.com/PossiblyAShrub/0aea9511b84c34e191eaa90dd7225969
208 static ImGuiDockNodeFlags dockspace_flags = ImGuiDockNodeFlags_PassthruCentralNode | ImGuiDockNodeFlags_NoWindowMenuButton;
209
210 ImGuiWindowFlags window_flags = ImGuiWindowFlags_MenuBar | ImGuiWindowFlags_NoDocking;
211 ImGui::SetNextWindowPos(imguiViewport->Pos);
212 ImGui::SetNextWindowSize(imguiViewport->Size);
213 ImGui::SetNextWindowViewport(imguiViewport->ID);
214 ImGui::PushStyleVar(ImGuiStyleVar_WindowRounding, 0.0f);
215 ImGui::PushStyleVar(ImGuiStyleVar_WindowBorderSize, 0.0f);
216 window_flags |= ImGuiWindowFlags_NoTitleBar | ImGuiWindowFlags_NoCollapse | ImGuiWindowFlags_NoResize | ImGuiWindowFlags_NoMove;
217 window_flags |= ImGuiWindowFlags_NoBringToFrontOnFocus | ImGuiWindowFlags_NoNavFocus;
218 if (dockspace_flags & ImGuiDockNodeFlags_PassthruCentralNode) {
219 window_flags |= ImGuiWindowFlags_NoBackground;
220 }
221
222 ImGui::PushStyleVar(ImGuiStyleVar_WindowPadding, ImVec2(0.0f, 0.0f));
223 ImGui::Begin("DockSpace", nullptr, window_flags);
224 ImGui::PopStyleVar();
225 ImGui::PopStyleVar(2);
226
227 const ImGuiIO& io = ImGui::GetIO();
228 if (io.ConfigFlags & ImGuiConfigFlags_DockingEnable) {
229 ImGuiID dockspace_id = ImGui::GetID("MyDockSpace");
230 // Check for saved dockspace settings
231 static bool iniLoaded = ImGui::DockBuilderGetNode(dockspace_id) != nullptr;
232
233 ImGui::DockSpace(dockspace_id, ImVec2(0.0f, 0.0f), dockspace_flags);
234
235 // Make initial dockspace if not previous settings available
236 if (!iniLoaded) {
237 iniLoaded = true;
238
239 ImGui::DockBuilderRemoveNode(dockspace_id); // clear any previous layout
240 ImGui::DockBuilderAddNode(dockspace_id, dockspace_flags | ImGuiDockNodeFlags_DockSpace);
241 ImGui::DockBuilderSetNodeSize(dockspace_id, imguiViewport->Size);
242
243 // Return value = pointer param set to nullptr
244 // v returns right side split
245 auto right_dock_id = ImGui::DockBuilderSplitNode(dockspace_id, ImGuiDir_Right, 0.25f, nullptr, &dockspace_id);
246 const auto rDown_dock_id = ImGui::DockBuilderSplitNode(right_dock_id, ImGuiDir_Down, 0.4f, nullptr, &right_dock_id);
247 }
248 }
249
250 ImGui::End();
