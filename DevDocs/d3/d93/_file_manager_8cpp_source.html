442bool GEU::FileManager::loadPointCloud(const std::string& filepath, const bool useClassification, const bool optimizeCloud,
443 const bool asyncOptimize) {
444 const auto filePath = std::filesystem::path(filepath);
445 if (exists(filePath)) {
446 std::string extension = filePath.extension().string();
447 if (extension == ".ply")
448 return loadPly(filePath.string(), useClassification, optimizeCloud, asyncOptimize);
449 if (extension == ".las")
450 return loadLas(filePath.string(), useClassification, optimizeCloud, asyncOptimize);
451 if (extension == ".laz") //El metodo LAS comprueba si está comprimido o no, por eso se usa el mismo.
452 return loadLas(filePath.string(), useClassification, optimizeCloud, asyncOptimize);
453
454 spdlog::warn("Cannot load {} file", extension);
455 } else {
456 spdlog::error("{} doesn't exist", filePath.filename().string());
457 }
458 return false;
459}
